Basic Toasts

toast.success('Message', 'Title'); toast.error('Message', 'Title'); toast.warning('Message', 'Title'); toast.info('Message', 'Title');

Custom Duration

toast.info('Message', 'Title', 2000); toast.success('Message', 'Title', 5000); toast.warning('Message', 'Title', 10000); toast.error('Message', 'Title', 0);

Confirm Dialogs

toast.confirm({ title: 'Confirm', message: 'Are you sure?', type: 'warning' }).then(result => { if (result) { // Confirmed } });

Prompt Dialogs

toast.prompt({ title: 'Input', message: 'Enter value:', placeholder: 'Type here...', type: 'info' }).then(value => { if (value !== null) { // User entered: value } });

Multiple Toasts

// Show multiple toast.success('First', 'Success'); toast.info('Second', 'Info'); toast.warning('Third', 'Warning'); // Clear all toast.clear();
